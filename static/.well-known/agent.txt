# ── MCP SURFACE ───────────────────────────────────────────
# Connect any MCP-compatible AI to XAP settlement capabilities.
# No account required for sandbox. ZexRail account required for live.

mcp_server: https://mcp.zexrail.com/sse
mcp_install_npm: npx -y @agenticamem/xap-mcp
mcp_install_python: pip install xap-sdk[mcp]
mcp_setup: python -m xap.mcp.setup
mcp_docs: https://zexrail.com/docs/mcp

mcp_tools:
  xap_discover_agents      Discover XAP agents by capability, price, success rate
  xap_verify_manifest      Verify agent trust credential (TSA, policy, attestation)
  xap_create_offer         Create a negotiation offer to an agent
  xap_respond_to_offer     Accept, reject, or counter a negotiation offer
  xap_settle               Execute a settlement with conditional hold
  xap_verify_receipt       Verify any XAP receipt (public, no auth)
  xap_check_balance        Check sandbox or live balance
  xap_verify_workflow      Verify complete causal chain of multi-agent workflow

# ── SEVEN TRUST PROPERTIES ────────────────────────────────
# Every VerityReceipt produced by ZexRail carries up to 7 properties:
# 1. Existence        — permanently recorded
# 2. Integrity        — hash chain valid
# 3. Correctness      — policy version hash-verified
# 4. Determinism      — deterministically replayable
# 5. Attribution      — signed with auditable key history
# 6. Causality        — position in workflow chain navigable
# 7. Third-party      — external verifiers attested their results
trust_properties: 7
trust_properties_url: https://xap-protocol.org/spec#trust-properties
